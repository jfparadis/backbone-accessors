Examples of accessors


            var Model = Backbone.Model.extend({

                defaults: {
                    firstName   : 'Jack',
                    lastName    : 'Nobody'
                },

            });

            Backbone.accessors.apply(Model);

            var model = new Model();

            model.getFirstName();
